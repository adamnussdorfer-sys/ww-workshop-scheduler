**`src/context/AppContext.jsx`:**
Create a React context for shared state. Export `AppContext` and a `useApp` custom hook that throws if used outside the provider. The context will hold `{ coaches, setCoaches, workshops, setWorkshops }` — but the actual state values are set in App.jsx (Plan 03 will provide real data).

```jsx
import { createContext, useContext } from 'react';

export const AppContext = createContext(null);

export function useApp() {
  const ctx = useContext(AppContext);
  if (!ctx) throw new Error('useApp must be used within AppContext.Provider');
  return ctx;
}
```

**`src/components/layout/AppShell.jsx`:**
CSS Grid layout with two columns (sidebar + main) and two rows (topbar + content). The sidebar column width is dynamic based on collapsed state. Use `useState` for `sidebarCollapsed`. Pass collapsed state and toggle to Sidebar. Grid must use inline `style` for `gridTemplateColumns` because the value is dynamic (Tailwind JIT cannot handle runtime values).

```jsx
import { useState } from 'react';
import Sidebar from './Sidebar';
import TopBar from './TopBar';

export default function AppShell({ children }) {
  const [sidebarCollapsed, setSidebarCollapsed] = useState(false);
  const sidebarWidth = sidebarCollapsed ? '64px' : '240px';

  return (
    <div
      className="h-screen grid overflow-hidden"
      style={{
        gridTemplateColumns: `${sidebarWidth} 1fr`,
        gridTemplateRows: '56px 1fr',
      }}
    >
      <Sidebar
        collapsed={sidebarCollapsed}
        onToggle={() => setSidebarCollapsed(prev => !prev)}
      />
      <TopBar />
      <main className="overflow-auto bg-surface p-6 col-start-2 row-start-2">
        {children}
      </main>
    </div>
  );
}
```
